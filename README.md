### docker-compose

```
version: '3.7'
services:
  veso:
    container_name: veso
    environment:
      - TZ=America/New_York
    user: 1000:1000
    volumes:
      - '/dev/dri:/dev/dri'
      - '/path/to/config:/config'
      - '/path/to/cache:/cache'
      - '/path/to/media:/media'
    ports:
      - 8096:8096
      - 8920:8920
    devices:
      - /dev/dri:/dev/dri
    restart: unless-stopped
    image: vesotv/veso:latest
